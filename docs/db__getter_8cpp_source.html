98 uint32_t database::last_non_undoable_block_num() const
99 {
100  //see https://github.com/bitshares/bitshares-core/issues/377
101  /*
102  There is a case when a value of undo_db.size() is greater then head_block_num(),
103  and as result we get a wrong value for last_non_undoable_block_num.
104  To resolve it we should take into account a number of active_sessions in calculations of
105  last_non_undoable_block_num (active sessions are related to a new block which is under generation).
106  */
107  return head_block_num() - ( _undo_db.size() - _undo_db.active_sessions() );
108 }
