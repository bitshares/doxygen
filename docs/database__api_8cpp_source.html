190  // Note:
191  // Different type of object_id<T> objects could become identical after packed.
192  // For example, both `account_id_type a=1.2.0` and `asset_id_type b=1.3.0` will become `0` after packed.
193  // In order to avoid collision, we don't use a template function here, instead, we implicitly convert all
194  // object IDs to `object_id_type` when subscribing.
195  //
196  // If need to subscribe to other data types, override this function with the types as parameter.
197  // For example, we had a `get_subscription_key( const public_key_type& item )` function here, which was
198  // removed lately since we no longer subscribe to public keys.
199  vector<char> get_subscription_key( const object_id_type& item )const
200  {
201  return fc::raw::pack(item);
202  }

299  {
300  uint64_t api_limit_get_limit_orders=_app_options->api_limit_get_limit_orders;
301  FC_ASSERT( limit <= api_limit_get_limit_orders );
302 
303  const auto& limit_order_idx = _db.get_index_type<limit_order_index>();
304  const auto& limit_price_idx = limit_order_idx.indices().get<by_price>();
305 
306  vector<limit_order_object> result;
307  result.reserve(limit*2);
308 
309  uint32_t count = 0;
310  auto limit_itr = limit_price_idx.lower_bound(price::max(a,b));
311  auto limit_end = limit_price_idx.upper_bound(price::min(a,b));
312  while(limit_itr != limit_end && count < limit)
313  {
314  result.push_back(*limit_itr);
315  ++limit_itr;
316  ++count;
317  }
318  count = 0;
319  limit_itr = limit_price_idx.lower_bound(price::max(b,a));
320  limit_end = limit_price_idx.upper_bound(price::min(b,a));
321  while(limit_itr != limit_end && count < limit)
322  {
323  result.push_back(*limit_itr);
324  ++limit_itr;
325  ++count;
326  }
327 
328  return result;

421  const fc::time_point_sec& now,
422  const asset_object& asset_base,
423  const asset_object& asset_quote,
424  const order_book& orders)
425 {
426  time = now;
427  base = asset_base.symbol;
428  quote = asset_quote.symbol;
429  percent_change = "0";
430  lowest_ask = "0";
431  highest_bid = "0";
432 
433  fc::uint128 bv;
434  fc::uint128 qv;
435  price latest_price = asset( mto.latest_base, mto.base ) / asset( mto.latest_quote, mto.quote );
436  if( mto.base != asset_base.id )
437  latest_price = ~latest_price;
438  latest = database_api_impl::price_to_string( latest_price, asset_base, asset_quote );
439  if( mto.last_day_base != 0 && mto.last_day_quote != 0 // has trade data before 24 hours
440  && ( mto.last_day_base != mto.latest_base || mto.last_day_quote != mto.latest_quote ) ) // price changed
441  {
442  price last_day_price = asset( mto.last_day_base, mto.base ) / asset( mto.last_day_quote, mto.quote );
443  if( mto.base != asset_base.id )
444  last_day_price = ~last_day_price;
445  percent_change = price_diff_percent_string( last_day_price, latest_price );
446  }
447  if( asset_base.id == mto.base )
448  {
449  bv = mto.base_volume;
450  qv = mto.quote_volume;
451  }
452  else
453  {
454  bv = mto.quote_volume;
455  qv = mto.base_volume;
456  }
457  base_volume = uint128_amount_to_string( bv, asset_base.precision );
458  quote_volume = uint128_amount_to_string( qv, asset_quote.precision );
459 
460  if(!orders.asks.empty())
461  lowest_ask = orders.asks[0].price;
462  if(!orders.bids.empty())
463  highest_bid = orders.bids[0].price;
464 }

852  FC_ASSERT( assets[1], "Invalid quote asset symbol: ${s}", ("s",quote) );
853 
854  auto base_id = assets[0]->id;
855  auto quote_id = assets[1]->id;
856

859  FC_ASSERT(ostart_price->quote.asset_id == quote_id, "Quote asset inconsistent with start price");

867  if ( !ostart_id.valid() && !ostart_price.valid() )
868  {
869  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, price::max(base_id, quote_id)));

871  else if ( ostart_id.valid() )
872  {
873  // in case of the order been deleted during page querying
874  const limit_order_object *p_loo = _db.find(*ostart_id);
875 
876  if ( !p_loo )
877  {
878  if ( ostart_price.valid() )
879  {
880  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, *ostart_price, *ostart_id));

883  {

885  FC_THROW("Order id invalid (maybe just been canceled?), and start price not provided");
886  }
887  }
888  else

890  const limit_order_object &loo = *p_loo;

898  }
899  }
900  else
901  {
902  // if reach here start_price must be valid
903  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, *ostart_price));
904  }
905 
906  upper_itr = index_by_account.upper_bound(std::make_tuple(account->id, price::min(base_id, quote_id)));
907 
908  // Add the account's orders
909  for ( ; lower_itr != upper_itr && count < limit; ++lower_itr, ++count)
910  {
911  const limit_order_object &order = *lower_itr;
912  results.emplace_back(order);
913  }
