80 void account_history_plugin_impl::update_account_histories( const signed_block& b )
81 {
82  graphene::chain::database& db = database();
83  const vector<optional< operation_history_object > >& hist = db.get_applied_operations();
84  bool is_first = true;
85  auto skip_oho_id = [&is_first,&db,this]() {
86  if( is_first && db._undo_db.enabled() ) // this ensures that the current id is rolled back on undo
87  {
88  db.remove( db.create<operation_history_object>( []( operation_history_object& obj) {} ) );
89  is_first = false;
90  }
91  else
92  _oho_index->use_next_id();
93  };
94 
95  for( const optional< operation_history_object >& o_op : hist )
96  {
97  optional<operation_history_object> oho;
98 
99  auto create_oho = [&]() {
100  is_first = false;
101  return optional<operation_history_object>( db.create<operation_history_object>( [&]( operation_history_object& h )
102  {
103  if( o_op.valid() )
104  {
105  h.op = o_op->op;
106  h.result = o_op->result;
107  h.block_num = o_op->block_num;
108  h.trx_in_block = o_op->trx_in_block;
109  h.op_in_trx = o_op->op_in_trx;
110  h.virtual_op = o_op->virtual_op;
111  }
112  } ) );
113  };
114 
115  if( !o_op.valid() || ( _max_ops_per_account == 0 && _partial_operations ) )
116  {
117  // Note: the 2nd and 3rd checks above are for better performance, when the db is not clean,
118  // they will break consistency of account_stats.total_ops and removed_ops and most_recent_op
119  skip_oho_id();
120  continue;
121  }
122  else if( !_partial_operations )
123  // add to the operation history index
124  oho = create_oho();
125 
126  const operation_history_object& op = *o_op;
127 
128  // get the set of accounts this operation applies to
129  flat_set<account_id_type> impacted;
130  vector<authority> other;
131  // fee payer is added here
132  operation_get_required_authorities( op.op, impacted, impacted, other,
133  MUST_IGNORE_CUSTOM_OP_REQD_AUTHS( db.head_block_time() ) );
134 
135  if( op.op.is_type< account_create_operation >() )
136  impacted.insert( op.result.get<object_id_type>() );
137  else
138  operation_get_impacted_accounts( op.op, impacted,
139  MUST_IGNORE_CUSTOM_OP_REQD_AUTHS( db.head_block_time() ) );
140 
141  for( auto& a : other )
142  for( auto& item : a.account_auths )
143  impacted.insert( item.first );
144 
145  // be here, either _max_ops_per_account > 0, or _partial_operations == false, or both
146  // if _partial_operations == false, oho should have been created above
147  // so the only case should be checked here is:
148  // whether need to create oho if _max_ops_per_account > 0 and _partial_operations == true
149 
150  // for each operation this account applies to that is in the config link it into the history
151  if( _tracked_accounts.size() == 0 ) // tracking all accounts
152  {
153  // if tracking all accounts, when impacted is not empty (although it will always be),
154  // still need to create oho if _max_ops_per_account > 0 and _partial_operations == true
155  // so always need to create oho if not done
156  if (!impacted.empty() && !oho.valid()) { oho = create_oho(); }
157 
158  if( _max_ops_per_account > 0 )
159  {
160  // Note: the check above is for better performance, when the db is not clean,
161  // it breaks consistency of account_stats.total_ops and removed_ops and most_recent_op,
162  // but it ensures it's safe to remove old entries in add_account_history(...)
163  for( auto& account_id : impacted )
164  {
165  // we don't do index_account_keys here anymore, because
166  // that indexing now happens in observers' post_evaluate()
167 
168  // add history
169  add_account_history( account_id, oho->id );
170  }
171  }
172  }
173  else // tracking a subset of accounts
174  {
175  // whether need to create oho if _max_ops_per_account > 0 and _partial_operations == true ?
176  // the answer: only need to create oho if a tracked account is impacted and need to save history
177 
178  if( _max_ops_per_account > 0 )
179  {
180  // Note: the check above is for better performance, when the db is not clean,
181  // it breaks consistency of account_stats.total_ops and removed_ops and most_recent_op,
182  // but it ensures it's safe to remove old entries in add_account_history(...)
183  for( auto account_id : _tracked_accounts )
184  {
185  if( impacted.find( account_id ) != impacted.end() )
186  {
187  if (!oho.valid()) { oho = create_oho(); }
188  // add history
189  add_account_history( account_id, oho->id );
190  }
191  }
192  }
193  }
194  if (_partial_operations && ! oho.valid())
195  skip_oho_id();
196  }

199 void account_history_plugin_impl::add_account_history( const account_id_type account_id, const operation_history_id_type op_id )
200 {
201  graphene::chain::database& db = database();
202  const auto& stats_obj = account_id(db).statistics(db);
203  // add new entry
204  const auto& ath = db.create<account_transaction_history_object>( [&]( account_transaction_history_object& obj ){
205  obj.operation_id = op_id;
206  obj.account = account_id;
207  obj.sequence = stats_obj.total_ops + 1;
208  obj.next = stats_obj.most_recent_op;
209  });
210  db.modify( stats_obj, [&]( account_statistics_object& obj ){
211  obj.most_recent_op = ath.id;
212  obj.total_ops = ath.sequence;
213  });
214  // Amount of history to keep depends on if account is in the "extended history" list
215  bool extended_hist = false;
216  for ( auto eh_account_id : _extended_history_accounts ) {
217  extended_hist |= (account_id == eh_account_id);
218  }
219  if ( _extended_history_registrars.size() > 0 ) {
220  const account_id_type registrar_id = account_id(db).registrar;
221  for ( auto eh_registrar_id : _extended_history_registrars ) {
222  extended_hist |= (registrar_id == eh_registrar_id);
223  }
224  }
225  // _max_ops_per_account is guaranteed to be non-zero outside; max_ops_to_keep
226  // will likewise be non-zero, and also non-negative (it is unsigned).
227  auto max_ops_to_keep = _max_ops_per_account;
228  if (extended_hist && _extended_max_ops_per_account > max_ops_to_keep) {
229  max_ops_to_keep = _extended_max_ops_per_account;
230  }
231  // Remove the earliest account history entry if too many.
232  if( stats_obj.total_ops - stats_obj.removed_ops > max_ops_to_keep )
233  {
234  // look for the earliest entry
235  const auto& his_idx = db.get_index_type<account_transaction_history_index>();
236  const auto& by_seq_idx = his_idx.indices().get<by_seq>();
237  auto itr = by_seq_idx.lower_bound( boost::make_tuple( account_id, 0 ) );
238  // make sure don't remove the one just added
239  if( itr != by_seq_idx.end() && itr->account == account_id && itr->id != ath.id )
240  {
241  // if found, remove the entry, and adjust account stats object
242  const auto remove_op_id = itr->operation_id;
243  const auto itr_remove = itr;
244  ++itr;
245  db.remove( *itr_remove );
246  db.modify( stats_obj, [&]( account_statistics_object& obj ){
247  obj.removed_ops = obj.removed_ops + 1;
248  });
249  // modify previous node's next pointer
250  // this should be always true, but just have a check here
251  if( itr != by_seq_idx.end() && itr->account == account_id )
252  {
253  db.modify( *itr, [&]( account_transaction_history_object& obj ){
254  obj.next = account_transaction_history_id_type();
255  });
256  }
257  // else need to modify the head pointer, but it shouldn't be true
258 
259  // remove the operation history entry (1.11.x) if configured and no reference left
260  if( _partial_operations )
261  {
262  // check for references
263  const auto& by_opid_idx = his_idx.indices().get<by_opid>();
264  if( by_opid_idx.find( remove_op_id ) == by_opid_idx.end() )
265  {
266  // if no reference, remove
267  db.remove( remove_op_id(db) );
268  }
269  }
270  }
271  }

319 void account_history_plugin::plugin_initialize(const boost::program_options::variables_map& options)
320 {
321  database().applied_block.connect( [&]( const signed_block& b){ my->update_account_histories(b); } );
322  my->_oho_index = database().add_index< primary_index< operation_history_index > >();
323  database().add_index< primary_index< account_transaction_history_index > >();
324 
325  LOAD_VALUE_SET(options, "track-account", my->_tracked_accounts, graphene::chain::account_id_type);
326  if (options.count("partial-operations") > 0) {
327  my->_partial_operations = options["partial-operations"].as<bool>();
328  }
329  if (options.count("max-ops-per-account") > 0) {
330  my->_max_ops_per_account = options["max-ops-per-account"].as<uint64_t>();
331  }
332  if (options.count("extended-max-ops-per-account") > 0) {
333  auto emopa = options["extended-max-ops-per-account"].as<uint64_t>();
334  my->_extended_max_ops_per_account = (emopa > my->_max_ops_per_account) ? emopa : my->_max_ops_per_account;
335  }
336  LOAD_VALUE_SET(options, "extended-history-by-account", my->_extended_history_accounts,
337  graphene::chain::account_id_type);
338  LOAD_VALUE_SET(options, "extended-history-by-registrar", my->_extended_history_registrars,
339  graphene::chain::account_id_type);
340 }
