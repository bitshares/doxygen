83 void object_database::flush()
84 {
85  const auto tmp_dir = _data_dir / "object_database.tmp";
86  const auto old_dir = _data_dir / "object_database.old";
87  const auto target_dir = _data_dir / "object_database";
88 
89  if( fc::exists( tmp_dir ) )
90  fc::remove_all( tmp_dir );
91  fc::create_directories( tmp_dir / "lock" );
92  std::vector<fc::future<void>> tasks;
93  constexpr size_t max_tasks = 200;
94  tasks.reserve(max_tasks);
95 
96  auto push_task = [this,&tasks,&tmp_dir]( size_t space, size_t type ) {
97  if( _index[space][type] )
98  tasks.push_back( fc::do_parallel( [this,space,type,&tmp_dir] () {
99  _index[space][type]->save( tmp_dir / fc::to_string(space) / fc::to_string(type) );
100  } ) );
101  };
102 
103  const auto spaces = _index.size();
104  for( size_t space = 0; space < spaces; ++space )
105  {
106  fc::create_directories( tmp_dir / fc::to_string(space) );
107  const auto types = _index[space].size();
108  for( size_t type = 0; type < types; ++type )
109  push_task( space, type );
110  }
111  for( auto& task : tasks )
112  task.wait();
113  fc::remove_all( tmp_dir / "lock" );
114  if( fc::exists( target_dir ) )
115  {
116  if( fc::exists( old_dir ) )
117  fc::remove_all( old_dir );
118  fc::rename( target_dir, old_dir );
119  }
120  fc::rename( tmp_dir, target_dir );
121  fc::remove_all( old_dir );
122 }

147  } ) );
148  };
149 
150  ilog("Opening object database from ${d} ...", ("d", data_dir));
151  const auto spaces = _index.size();

153  {
154  const auto types = _index[space].size();
