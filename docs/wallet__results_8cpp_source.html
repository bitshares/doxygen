44  auto format_account_history = [this](variant result, const fc::variants&)
45  {
46  auto r = result.as<vector<operation_detail>>( GRAPHENE_MAX_NESTED_OBJECTS );

48

51  operation_history_object& i = d.op;

149  m["get_order_book"] = [](variant result, const fc::variants&)
150  {
151  auto orders = result.as<order_book>( GRAPHENE_MAX_NESTED_OBJECTS );
152  auto bids = orders.bids;

157  double bid_sum = 0;
158  double ask_sum = 0;
159  const int spacing = 20;
160 
161  auto prettify_num = [&ss]( double n )
162  {
163  if (abs( round( n ) - n ) < 0.00000000001 )
164  {
165  ss << (int) n;
166  }
167  else if (n - floor(n) < 0.000001)
168  {
169  ss << setiosflags( ios::fixed ) << setprecision(10) << n;
170  }
171  else
172  {
173  ss << setiosflags( ios::fixed ) << setprecision(6) << n;
174  }
175  };
176  auto prettify_num_string = [&]( string& num_string )
177  {
178  double n = fc::to_double( num_string );
179  prettify_num( n );
180  };
181 
182  ss << setprecision( 8 ) << setiosflags( ios::fixed ) << setiosflags( ios::left );
183 
184  ss << ' ' << setw( (spacing * 4) + 6 ) << "BUY ORDERS" << "SELL ORDERS\n"
185  << ' ' << setw( spacing + 1 ) << "Price" << setw( spacing ) << orders.quote << ' ' << setw( spacing )
186  << orders.base << ' ' << setw( spacing ) << sum_stream.str()
187  << " " << setw( spacing + 1 ) << "Price" << setw( spacing ) << orders.quote << ' ' << setw( spacing )
188  << orders.base << ' ' << setw( spacing ) << sum_stream.str()
189  << "\n====================================================================================="
190  << "|=====================================================================================\n";
191 
192  for (unsigned int i = 0; i < bids.size() || i < asks.size() ; i++)
193  {
194  if ( i < bids.size() )
195  {
196  bid_sum += fc::to_double( bids[i].base );
197  ss << ' ' << setw( spacing );
198  prettify_num_string( bids[i].price );
199  ss << ' ' << setw( spacing );
200  prettify_num_string( bids[i].quote );
201  ss << ' ' << setw( spacing );
202  prettify_num_string( bids[i].base );
203  ss << ' ' << setw( spacing );
204  prettify_num( bid_sum );
205  ss << ' ';
206  }
207  else
208  {
209  ss << setw( (spacing * 4) + 5 ) << ' ';
210  }
211 
212  ss << '|';
213 
214  if ( i < asks.size() )
215  {
216  ask_sum += fc::to_double( asks[i].base );
217  ss << ' ' << setw( spacing );
218  prettify_num_string( asks[i].price );
219  ss << ' ' << setw( spacing );
220  prettify_num_string( asks[i].quote );
221  ss << ' ' << setw( spacing );
222  prettify_num_string( asks[i].base );
223  ss << ' ' << setw( spacing );
224  prettify_num( ask_sum );
225  }
226 
227  ss << '\n';
228  }
229 
230  ss << endl
231  << "Buy Total: " << bid_sum << ' ' << orders.base << endl
232  << "Sell Total: " << ask_sum << ' ' << orders.base << endl;
233 
234  return ss.str();
235  };
