96 bool handleBulkResponse(long http_code, const std::string& CurlReadBuffer)
97 {
98  if(http_code == 200) {
99  // all good, but check errors in response
100  fc::variant j = fc::json::from_string(CurlReadBuffer);
101  bool errors = j["errors"].as_bool();
102  if(errors == true) {
103  elog( "ES returned 200 but with errors: ${e}", ("e", CurlReadBuffer) );
104  return false;
105  }
106  }
107  else {
108  if(http_code == 413) {
109  elog( "413 error: Can be low disk space. ${e}", ("e", CurlReadBuffer) );
110  }
111  else if(http_code == 401) {
112  elog( "401 error: Unauthorized. ${e}", ("e", CurlReadBuffer) );
113  }
114  else {
115  elog( "${code} error: ${e}", ("code", std::to_string(http_code)) ("e", CurlReadBuffer) );
116  }
117  return false;
118  }
119  return true;
120 }

167 const std::string doCurl(CurlRequest& curl)
168 {
169  std::string CurlReadBuffer;
170  struct curl_slist *headers = NULL;
171  headers = curl_slist_append(headers, "Content-Type: application/json");
172 
173  curl_easy_setopt(curl.handler, CURLOPT_HTTPHEADER, headers);
174  curl_easy_setopt(curl.handler, CURLOPT_URL, curl.url.c_str());
175  curl_easy_setopt(curl.handler, CURLOPT_CUSTOMREQUEST, curl.type.c_str());
176  if(curl.type == "POST")
177  {
178  curl_easy_setopt(curl.handler, CURLOPT_POST, true);
179  curl_easy_setopt(curl.handler, CURLOPT_POSTFIELDS, curl.query.c_str());
180  }
181  curl_easy_setopt(curl.handler, CURLOPT_WRITEFUNCTION, WriteCallback);
182  curl_easy_setopt(curl.handler, CURLOPT_WRITEDATA, (void *)&CurlReadBuffer);
183  curl_easy_setopt(curl.handler, CURLOPT_USERAGENT, "libcrp/0.1");
184  if(!curl.auth.empty())
185  curl_easy_setopt(curl.handler, CURLOPT_USERPWD, curl.auth.c_str());
186  curl_easy_perform(curl.handler);
187 
188  return CurlReadBuffer;
189 }
