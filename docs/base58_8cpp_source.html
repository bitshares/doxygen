7 //
8 // Why base-58 instead of standard base-64 encoding?
9 // - Don't want 0OIl characters that look the same in some fonts and
10 // could be used to create visually identical looking account numbers.
11 // - A string with non-alphanumeric characters is not as easily accepted as an account number.
12 // - E-mail usually won't line-break if there's no punctuation to break at.
13 // - Doubleclicking selects the whole number as one word if it's all alphanumeric.
14 //

138  void setint64(int64_t n)
139  {
140  unsigned char pch[sizeof(n) + 6];
141  unsigned char* p = pch + 4;
142  bool fNegative = false;
143  if (n < (int64_t)0)
144  {
145  n = -n;
146  fNegative = true;
147  }
148  bool fLeadingZeroes = true;
149  for (int i = 0; i < 8; i++)
150  {
151  unsigned char c = (n >> 56) & 0xff;
152  n <<= 8;
153  if (fLeadingZeroes)
154  {
155  if (c == 0)
156  continue;
157  if (c & 0x80)
158  *p++ = (fNegative ? 0x80 : 0);
159  else if (fNegative)
160  c |= 0x80;
161  fLeadingZeroes = false;
162  }
163  *p++ = c;
164  }
165  unsigned int nSize = p - (pch + 4);
166  pch[0] = (nSize >> 24) & 0xff;
167  pch[1] = (nSize >> 16) & 0xff;
168  pch[2] = (nSize >> 8) & 0xff;
169  pch[3] = (nSize) & 0xff;
170  BN_mpi2bn(pch, p - pch, bn);
171  }

201  void setvch(const std::vector<unsigned char>& vch)
202  {
203  std::vector<unsigned char> vch2(vch.size() + 4);
204  unsigned int nSize = vch.size();
205  // BIGNUM's byte stream format expects 4 bytes of
206  // big endian size data info at the front
207  vch2[0] = (nSize >> 24) & 0xff;
208  vch2[1] = (nSize >> 16) & 0xff;
209  vch2[2] = (nSize >> 8) & 0xff;
210  vch2[3] = (nSize >> 0) & 0xff;
211  // swap data to big endian
212  reverse_copy(vch.begin(), vch.end(), vch2.begin() + 4);
213  BN_mpi2bn(&vch2[0], vch2.size(), bn);
214  }

253  void SetHex(const std::string& str)
254  {
255  // skip 0x
256  const char* psz = str.c_str();
257  while (isspace(*psz))
258  psz++;
259  bool fNegative = false;
260  if (*psz == '-')
261  {
262  fNegative = true;
263  psz++;
264  }
265  if (psz[0] == '0' && tolower(psz[1]) == 'x')
266  psz += 2;
267  while (isspace(*psz))
268  psz++;
269

273  while (isxdigit(*psz))
274  {
275  *this <<= 4;

277  *this += n;

284  {
285  CAutoBN_CTX pctx;
286  CBigNum bnBase = nBase;
287  CBigNum bn0 = 0;
288  std::string str;
289  CBigNum bn = *this;
290  BN_set_negative(bn.bn, false);
291  CBigNum dv;
292  CBigNum rem;

294  return "0";

296  {
297  if (!BN_div(dv.bn, rem.bn, bn.bn, bnBase.bn, pctx))
298  throw bignum_error("CBigNum::ToString() : BN_div failed");

300  unsigned int c = rem.getulong();
301  str += "0123456789abcdef"[c];
302  }
303  if (BN_is_negative(this->bn))
304  str += "-";
305  reverse(str.begin(), str.end());
306  return str;

503 // Encode a byte sequence as a base58-encoded string
504 inline std::string EncodeBase58(const unsigned char* pbegin, const unsigned char* pend)
505 {
506  CAutoBN_CTX pctx;
507  CBigNum bn58 = 58;
508  CBigNum bn0 = 0;
509 
510  // Convert big endian data to little endian
511  // Extra zero at the end make sure bignum will interpret as a positive number
512  std::vector<unsigned char> vchTmp(pend-pbegin+1, 0);
513  reverse_copy(pbegin, pend, vchTmp.begin());
514 
515  // Convert little endian data to bignum
516  CBigNum bn;
517  bn.setvch(vchTmp);
518 
519  // Convert bignum to std::string
520  std::string str;
521  // Expected size increase from base58 conversion is approximately 137%
522  // use 138% to be safe
523  str.reserve((pend - pbegin) * 138 / 100 + 1);
524  CBigNum dv;
525  CBigNum rem;
526  while (bn > bn0)
527  {
528  if (!BN_div(dv.to_bignum(), rem.to_bignum(), bn.to_bignum(), bn58.to_bignum(), pctx))
529  throw bignum_error("EncodeBase58 : BN_div failed");
530  bn = dv;
531  unsigned int c = rem.getulong();
532  str += pszBase58[c];
533  }
534 
535  // Leading zeroes encoded as base58 zeros
536  for (const unsigned char* p = pbegin; p < pend && *p == 0; p++)
537  str += pszBase58[0];
538 
539  // Convert little endian std::string to big endian
540  reverse(str.begin(), str.end());
541 // slog( "Encode '%s'", str.c_str() );
542  return str;
543 }

551 // Decode a base58-encoded string psz into byte vector vchRet
552 // returns true if decoding is succesful
553 inline bool DecodeBase58(const char* psz, std::vector<unsigned char>& vchRet)
554 {
555  CAutoBN_CTX pctx;
556  vchRet.clear();
557  CBigNum bn58 = 58;
558  CBigNum bn = 0;
559  CBigNum bnChar;
560  while (isspace(*psz))
561  psz++;
562 
563  // Convert big endian string to bignum
564  for (const char* p = psz; *p; p++)
565  {
566  const char* p1 = strchr(pszBase58, *p);
567  if (p1 == NULL)
568  {
569  while (isspace(*p))
570  p++;
571  if (*p != '\0') {
572  //slog( "%s '%c'", pszBase58,*p );
573  return false;
574  }
575  break;
576  }
577  bnChar.setulong(p1 - pszBase58);
578  if (!BN_mul(bn.to_bignum(), bn.to_bignum(), bn58.to_bignum(), pctx))
579  throw bignum_error("DecodeBase58 : BN_mul failed");
580  bn += bnChar;
581  }
582 
583  // Get bignum as little endian data
584  std::vector<unsigned char> vchTmp = bn.getvch();
585 
586  // Trim off sign byte if present
587  if (vchTmp.size() >= 2 && vchTmp.end()[-1] == 0 && vchTmp.end()[-2] >= 0x80)
588  vchTmp.erase(vchTmp.end()-1);
589 
590  // Restore leading zeros
591  int nLeadingZeros = 0;
592  for (const char* p = psz; *p == pszBase58[0]; p++)
593  nLeadingZeros++;
594  vchRet.assign(nLeadingZeros + vchTmp.size(), 0);
595 
596  // Convert little endian data to big endian
597  reverse_copy(vchTmp.begin(), vchTmp.end(), vchRet.end() - vchTmp.size());
598  return true;
599 }
