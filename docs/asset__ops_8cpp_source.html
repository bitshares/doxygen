38 bool is_valid_symbol( const string& symbol )
39 {
40  static const std::locale& loc = std::locale::classic();
41  if( symbol.size() < GRAPHENE_MIN_ASSET_SYMBOL_LENGTH )
42  return false;
43 
44  if( symbol.substr(0,3) == "BIT" )
45  return false;
46 
47  if( symbol.size() > GRAPHENE_MAX_ASSET_SYMBOL_LENGTH )
48  return false;
49 
50  if( !isalpha( symbol.front(), loc ) )
51  return false;
52 
53  if( !isalnum( symbol.back(), loc ) )
54  return false;
55 
56  bool dot_already_present = false;
57  for( const auto c : symbol )
58  {
59  if( (isalpha( c, loc ) && isupper( c, loc )) || isdigit( c, loc ) )
60  continue;
61 
62  if( c == '.' )
63  {
64  if( dot_already_present )
65  return false;
66 
67  dot_already_present = true;
68  continue;
69  }
70 
71  return false;
72  }
73 
74  return true;
75 }

82 share_type asset_create_operation::calculate_fee(const asset_create_operation::fee_parameters_type& param)const
83 {
84  auto core_fee_required = param.long_symbol;
85 
86  switch(symbol.size()) {
87  case 3: core_fee_required = param.symbol3;
88  break;
89  case 4: core_fee_required = param.symbol4;
90  break;
91  default:
92  break;
93  }
94 
95  // common_options contains several lists and a string. Charge fees for its size
96  core_fee_required += calculate_data_fee( fc::raw::pack_size(*this), param.price_per_kbyte );
97 
98  return core_fee_required;
99 }

101 void asset_create_operation::validate()const
102 {
103  FC_ASSERT( fee.amount >= 0 );
104  FC_ASSERT( is_valid_symbol(symbol) );
105  common_options.validate();
106  if( common_options.issuer_permissions & (disable_force_settle|global_settle) )
107  FC_ASSERT( bitasset_opts.valid() );
108  if( is_prediction_market )
109  {
110  FC_ASSERT( bitasset_opts.valid(), "Cannot have a User-Issued Asset implement a prediction market." );
111  FC_ASSERT( common_options.issuer_permissions & global_settle );
112  }
113  if( bitasset_opts ) bitasset_opts->validate();
114 
115  asset dummy = asset(1) * common_options.core_exchange_rate;
116  FC_ASSERT(dummy.asset_id == asset_id_type(1));
117  FC_ASSERT(precision <= 12);
118 }

215 void asset_options::validate()const
216 {
217  FC_ASSERT( max_supply > 0 );
218  FC_ASSERT( max_supply <= GRAPHENE_MAX_SHARE_SUPPLY );
219  FC_ASSERT( market_fee_percent <= GRAPHENE_100_PERCENT );
220  FC_ASSERT( max_market_fee >= 0 && max_market_fee <= GRAPHENE_MAX_SHARE_SUPPLY );
221  // There must be no high bits in permissions whose meaning is not known.
222  FC_ASSERT( !(issuer_permissions & ~ASSET_ISSUER_PERMISSION_MASK) );
223  // The global_settle flag may never be set (this is a permission only)
224  FC_ASSERT( !(flags & global_settle) );
225  // the witness_fed and committee_fed flags cannot be set simultaneously
226  FC_ASSERT( (flags & (witness_fed_asset | committee_fed_asset)) != (witness_fed_asset | committee_fed_asset) );
227  core_exchange_rate.validate();
228  FC_ASSERT( core_exchange_rate.base.asset_id.instance.value == 0 ||
229  core_exchange_rate.quote.asset_id.instance.value == 0 );
230 
231  if(!whitelist_authorities.empty() || !blacklist_authorities.empty())
232  FC_ASSERT( flags & white_list );
233  for( auto item : whitelist_markets )
234  {
235  FC_ASSERT( blacklist_markets.find(item) == blacklist_markets.end() );
236  }
237  for( auto item : blacklist_markets )
238  {
239  FC_ASSERT( whitelist_markets.find(item) == whitelist_markets.end() );
240  }
241  if( extensions.value.reward_percent.valid() )
242  FC_ASSERT( *extensions.value.reward_percent < GRAPHENE_100_PERCENT );
243 }
