127 static bool equal( const fc::variant& a, const fc::variant& b )
128 {
129  auto a_type = a.get_type();
130  auto b_type = b.get_type();
131  if( a_type == fc::variant::type_id::int64_type && a.as<int64_t>(1) > 0 )
132  a_type = fc::variant::type_id::uint64_type;
133  if( b_type == fc::variant::type_id::int64_type && b.as<int64_t>(1) > 0 )
134  b_type = fc::variant::type_id::uint64_type;
135  if( a_type != b_type )
136  {
137  if( ( a_type == fc::variant::type_id::double_type
138  && b_type == fc::variant::type_id::string_type )
139  || ( a_type == fc::variant::type_id::string_type
140  && b_type == fc::variant::type_id::double_type ) )
141  return a.as<double>(1) == b.as<double>(1);
142  return false;
143  }
144  switch( a_type )
145  {
146  case fc::variant::type_id::null_type: return true;
147  case fc::variant::type_id::int64_type: return a.as<int64_t>(1) == b.as<int64_t>(1);
148  case fc::variant::type_id::uint64_type: return a.as<uint64_t>(1) == b.as<uint64_t>(1);
149  case fc::variant::type_id::double_type: return a.as<double>(1) == b.as<double>(1);
150  case fc::variant::type_id::bool_type: return a.as<bool>(1) == b.as<bool>(1);
151  case fc::variant::type_id::string_type: return a.as<std::string>(1) == b.as<std::string>(1);
152  case fc::variant::type_id::array_type:
153  if( a.get_array().size() != b.get_array().size() ) return false;
154  else
155  {
156  std::vector<fc::variant>::const_iterator b_it = b.get_array().begin();
157  for( const auto& a_it : a.get_array() )
158  {
159  if( !equal( a_it, *b_it ) ) return false;
160  b_it++;
161  }
162  }
163  return true;
164  case fc::variant::type_id::object_type:
165  if( a.get_object().size() != b.get_object().size() ) return false;
166  for( const auto& a_it : a.get_object() )
167  {
168  const auto& b_obj = b.get_object().find( a_it.key() );
169  if( b_obj == b.get_object().end() || !equal( a_it.value(), b_obj->value() ) ) return false;
170  }
171  return true;
172  case fc::variant::type_id::blob_type:
173  default:
174  FC_THROW_EXCEPTION( fc::invalid_arg_exception, "Unsupported variant type: {t}", ( "t", a.get_type() ) );
175  }
176 }
