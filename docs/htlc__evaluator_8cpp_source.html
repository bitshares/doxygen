38  void_result htlc_create_evaluator::do_evaluate(const htlc_create_operation& o)
39  {
40  graphene::chain::database& d = db();
41  optional<htlc_options> htlc_options = get_committee_htlc_options(db());
42 
43  FC_ASSERT(htlc_options, "HTLC Committee options are not set.");
44 
45  // make sure the expiration is reasonable
46  FC_ASSERT( o.claim_period_seconds <= htlc_options->max_timeout_secs, "HTLC Timeout exceeds allowed length" );
47  // make sure the preimage length is reasonable
48  FC_ASSERT( o.preimage_size <= htlc_options->max_preimage_size, "HTLC preimage length exceeds allowed length" );
49  // make sure the sender has the funds for the HTLC
50  FC_ASSERT( d.get_balance( o.from, o.amount.asset_id) >= (o.amount), "Insufficient funds") ;
51  const auto& asset_to_transfer = o.amount.asset_id( d );
52  const auto& from_account = o.from( d );
53  const auto& to_account = o.to( d );
54  FC_ASSERT( is_authorized_asset( d, from_account, asset_to_transfer ),
55  "Asset ${asset} is not authorized for account ${acct}.",
56  ( "asset", asset_to_transfer.id )( "acct", from_account.id ) );
57  FC_ASSERT( is_authorized_asset( d, to_account, asset_to_transfer ),
58  "Asset ${asset} is not authorized for account ${acct}.",
59  ( "asset", asset_to_transfer.id )( "acct", to_account.id ) );
60  return void_result();
61  }

63  object_id_type htlc_create_evaluator::do_apply(const htlc_create_operation& o)
64  {
65  try {
66  graphene::chain::database& dbase = db();
67  dbase.adjust_balance( o.from, -o.amount );
68 
69  const htlc_object& esc = db().create<htlc_object>([&dbase,&o]( htlc_object& esc ) {
70  esc.transfer.from = o.from;
71  esc.transfer.to = o.to;
72  esc.transfer.amount = o.amount.amount;
73  esc.transfer.asset_id = o.amount.asset_id;
74  esc.conditions.hash_lock.preimage_hash = o.preimage_hash;
75  esc.conditions.hash_lock.preimage_size = o.preimage_size;
76  esc.conditions.time_lock.expiration = dbase.head_block_time() + o.claim_period_seconds;
77  });
78  return esc.id;
79 
80  } FC_CAPTURE_AND_RETHROW( (o) )
81  }

112  void_result htlc_redeem_evaluator::do_apply(const htlc_redeem_operation& o)
113  {
114  db().adjust_balance(htlc_obj->transfer.to, asset(htlc_obj->transfer.amount, htlc_obj->transfer.asset_id) );
115  // notify related parties
116  htlc_redeemed_operation virt_op( htlc_obj->id, htlc_obj->transfer.from, htlc_obj->transfer.to,
117  o.redeemer, asset(htlc_obj->transfer.amount, htlc_obj->transfer.asset_id ) );
118  db().push_applied_operation( virt_op );
119  db().remove(*htlc_obj);
120  return void_result();

123  void_result htlc_extend_evaluator::do_evaluate(const htlc_extend_operation& o)
124  {
125  htlc_obj = &db().get<htlc_object>(o.htlc_id);
126  FC_ASSERT(o.update_issuer == htlc_obj->transfer.from, "HTLC may only be extended by its creator.");
127  optional<htlc_options> htlc_options = get_committee_htlc_options(db());
128  FC_ASSERT( htlc_obj->conditions.time_lock.expiration.sec_since_epoch()
129  + static_cast<uint64_t>(o.seconds_to_add) < fc::time_point_sec::maximum().sec_since_epoch(),
130  "Extension would cause an invalid date");
131  FC_ASSERT( htlc_obj->conditions.time_lock.expiration + o.seconds_to_add
132  <= db().head_block_time() + htlc_options->max_timeout_secs,
133  "Extension pushes contract too far into the future" );
134  return void_result();
135  }
