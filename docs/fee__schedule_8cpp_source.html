121  asset fee_schedule::calculate_fee( const operation& op )const
122  {
123  uint64_t required_fee = op.visit( calc_fee_visitor( *this, op ) );
124  if( scale != GRAPHENE_100_PERCENT )
125  {
126  auto scaled = fc::uint128_t(required_fee) * scale;
127  scaled /= GRAPHENE_100_PERCENT;
128  FC_ASSERT( scaled <= GRAPHENE_MAX_SHARE_SUPPLY,
129  "Required fee after scaling would exceed maximum possible supply" );
130  required_fee = static_cast<uint64_t>(scaled);
131  }
132  return asset( required_fee );
133  }

140  asset fee_schedule::set_fee( operation& op, const price& core_exchange_rate )const
141  {
142  auto f = calculate_fee( op, core_exchange_rate );
143  auto f_max = f;
144  for( int i=0; i<MAX_FEE_STABILIZATION_ITERATION; i++ )
145  {
146  op.visit( set_fee_visitor( f_max ) );
147  auto f2 = calculate_fee( op, core_exchange_rate );
148  if( f == f2 )
149  break;
150  f_max = std::max( f_max, f2 );
151  f = f2;
152  if( i == 0 )
153  {
154  // no need for warnings on later iterations
155  wlog( "set_fee requires multiple iterations to stabilize with core_exchange_rate ${p} on operation ${op}",
156  ("p", core_exchange_rate) ("op", op) );
157  }
158  }
159  return f_max;
160  }

162  void chain_parameters::validate()const
163  {
164  get_current_fees().validate();
165  FC_ASSERT( reserve_percent_of_fee <= GRAPHENE_100_PERCENT );
166  FC_ASSERT( network_percent_of_fee <= GRAPHENE_100_PERCENT );
167  FC_ASSERT( lifetime_referrer_percent_of_fee <= GRAPHENE_100_PERCENT );
168  FC_ASSERT( network_percent_of_fee + lifetime_referrer_percent_of_fee <= GRAPHENE_100_PERCENT );
169 
170  FC_ASSERT( block_interval >= GRAPHENE_MIN_BLOCK_INTERVAL );
171  FC_ASSERT( block_interval <= GRAPHENE_MAX_BLOCK_INTERVAL );
172  FC_ASSERT( block_interval > 0 );
173  FC_ASSERT( maintenance_interval > block_interval,
174  "Maintenance interval must be longer than block interval" );
175  FC_ASSERT( maintenance_interval % block_interval == 0,
176  "Maintenance interval must be a multiple of block interval" );
177  FC_ASSERT( maximum_transaction_size >= GRAPHENE_MIN_TRANSACTION_SIZE_LIMIT,
178  "Transaction size limit is too low" );
179  FC_ASSERT( maximum_block_size >= GRAPHENE_MIN_BLOCK_SIZE_LIMIT,
180  "Block size limit is too low" );
181  FC_ASSERT( maximum_time_until_expiration > block_interval,
182  "Maximum transaction expiration time must be greater than a block interval" );
183  FC_ASSERT( maximum_proposal_lifetime - committee_proposal_review_period > block_interval,
184  "Committee proposal review period must be less than the maximum proposal lifetime" );
185  }
