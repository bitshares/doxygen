81  asset fee_schedule::set_fee( operation& op, const price& core_exchange_rate )const
82  {
83  auto f = calculate_fee( op, core_exchange_rate );
84  auto f_max = f;
85  for( size_t i=0; i<MAX_FEE_STABILIZATION_ITERATION; i++ )
86  {
87  op.visit( set_fee_visitor( f_max ) );
88  auto f2 = calculate_fee( op, core_exchange_rate );
89  if( f == f2 )
90  break;
91  f_max = std::max( f_max, f2 );
92  f = f2;
93  if( i == 0 )
94  {
95  // no need for warnings on later iterations
96  wlog( "set_fee requires multiple iterations to stabilize with core_exchange_rate ${p} on operation ${op}",
97  ("p", core_exchange_rate) ("op", op) );
98  }
99  }
100  return f_max;
101  }
