298 void get_relevant_accounts( const object* obj, flat_set<account_id_type>& accounts )
299 {
300  if( obj->id.space() == protocol_ids )
301  {
302  switch( (object_type)obj->id.type() )
303  {
304  case null_object_type:
305  case base_object_type:
306  return;
307  case account_object_type:{
308  accounts.insert( obj->id );
309  break;
310  } case asset_object_type:{
311  const auto& aobj = dynamic_cast<const asset_object*>(obj);
312  FC_ASSERT( aobj != nullptr );
313  accounts.insert( aobj->issuer );
314  break;
315  } case force_settlement_object_type:{
316  const auto& aobj = dynamic_cast<const force_settlement_object*>(obj);
317  FC_ASSERT( aobj != nullptr );
318  accounts.insert( aobj->owner );
319  break;
320  } case committee_member_object_type:{
321  const auto& aobj = dynamic_cast<const committee_member_object*>(obj);
322  FC_ASSERT( aobj != nullptr );
323  accounts.insert( aobj->committee_member_account );
324  break;
325  } case witness_object_type:{
326  const auto& aobj = dynamic_cast<const witness_object*>(obj);
327  FC_ASSERT( aobj != nullptr );
328  accounts.insert( aobj->witness_account );
329  break;
330  } case limit_order_object_type:{
331  const auto& aobj = dynamic_cast<const limit_order_object*>(obj);
332  FC_ASSERT( aobj != nullptr );
333  accounts.insert( aobj->seller );
334  break;
335  } case call_order_object_type:{
336  const auto& aobj = dynamic_cast<const call_order_object*>(obj);
337  FC_ASSERT( aobj != nullptr );
338  accounts.insert( aobj->borrower );
339  break;
340  } case custom_object_type:{
341  break;
342  } case proposal_object_type:{
343  const auto& aobj = dynamic_cast<const proposal_object*>(obj);
344  FC_ASSERT( aobj != nullptr );
345  transaction_get_impacted_accounts( aobj->proposed_transaction, accounts );
346  break;
347  } case operation_history_object_type:{
348  const auto& aobj = dynamic_cast<const operation_history_object*>(obj);
349  FC_ASSERT( aobj != nullptr );
350  operation_get_impacted_accounts( aobj->op, accounts );
351  break;
352  } case withdraw_permission_object_type:{
353  const auto& aobj = dynamic_cast<const withdraw_permission_object*>(obj);
354  FC_ASSERT( aobj != nullptr );
355  accounts.insert( aobj->withdraw_from_account );
356  accounts.insert( aobj->authorized_account );
357  break;
358  } case vesting_balance_object_type:{
359  const auto& aobj = dynamic_cast<const vesting_balance_object*>(obj);
360  FC_ASSERT( aobj != nullptr );
361  accounts.insert( aobj->owner );
362  break;
363  } case worker_object_type:{
364  const auto& aobj = dynamic_cast<const worker_object*>(obj);
365  FC_ASSERT( aobj != nullptr );
366  accounts.insert( aobj->worker_account );
367  break;
368  } case balance_object_type:{
370  break;
371  } case htlc_object_type:{
372  const auto& htlc_obj = dynamic_cast<const htlc_object*>(obj);
373  FC_ASSERT( htlc_obj != nullptr );
374  accounts.insert( htlc_obj->transfer.from );
375  accounts.insert( htlc_obj->transfer.to );
376  break;
377  }
378  }
379  }
380  else if( obj->id.space() == implementation_ids )
381  {
382  switch( (impl_object_type)obj->id.type() )
383  {
384  case impl_global_property_object_type:
385  break;
386  case impl_dynamic_global_property_object_type:
387  break;
388  case impl_reserved0_object_type:
389  break;
390  case impl_asset_dynamic_data_object_type:
391  break;
392  case impl_asset_bitasset_data_object_type:
393  break;
394  case impl_account_balance_object_type:{
395  const auto& aobj = dynamic_cast<const account_balance_object*>(obj);
396  FC_ASSERT( aobj != nullptr );
397  accounts.insert( aobj->owner );
398  break;
399  } case impl_account_statistics_object_type:{
400  const auto& aobj = dynamic_cast<const account_statistics_object*>(obj);
401  FC_ASSERT( aobj != nullptr );
402  accounts.insert( aobj->owner );
403  break;
404  } case impl_transaction_history_object_type:{
405  const auto& aobj = dynamic_cast<const transaction_history_object*>(obj);
406  FC_ASSERT( aobj != nullptr );
407  transaction_get_impacted_accounts( aobj->trx, accounts );
408  break;
409  } case impl_blinded_balance_object_type:{
410  const auto& aobj = dynamic_cast<const blinded_balance_object*>(obj);
411  FC_ASSERT( aobj != nullptr );
412  for( const auto& a : aobj->owner.account_auths )
413  accounts.insert( a.first );
414  break;
415  } case impl_block_summary_object_type:
416  break;
417  case impl_account_transaction_history_object_type: {
418  const auto& aobj = dynamic_cast<const account_transaction_history_object*>(obj);
419  FC_ASSERT( aobj != nullptr );
420  accounts.insert( aobj->account );
421  break;
422  } case impl_chain_property_object_type:
423  break;
424  case impl_witness_schedule_object_type:
425  break;
426  case impl_budget_record_object_type:
427  break;
428  case impl_special_authority_object_type:
429  break;
430  case impl_buyback_object_type:
431  break;
432  case impl_fba_accumulator_object_type:
433  break;
434  case impl_collateral_bid_object_type:{
435  const auto& aobj = dynamic_cast<const collateral_bid_object*>(obj);
436  FC_ASSERT( aobj != nullptr );
437  accounts.insert( aobj->bidder );
438  break;
439  }
440  }
441  }
