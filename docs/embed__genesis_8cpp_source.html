79 void convert_to_c_array(
80  const std::string& src,
81  std::string& dest,
82  int width = 40 )
83 {
84  dest.reserve( src.length() * 6 / 5 );
85  bool needs_comma = false;
86  int row = 0;
87  for( std::string::size_type i=0; i<src.length(); i+=width )
88  {
89  std::string::size_type j = std::min( i+width, src.length() );
90  if( needs_comma )
91  dest.append(",\n");
92  dest.append("\"");
93  for( std::string::size_type k=i; k<j; k++ )
94  {
95  char c = src[k];
96  switch(c)
97  {
98  // use most short escape sequences
99  case '\"': dest.append("\\\""); break;
100  case '\?': dest.append("\\\?"); break;
101  case '\\': dest.append("\\\\"); break;
102  case '\a': dest.append( "\\a"); break;
103  case '\b': dest.append( "\\b"); break;
104  case '\f': dest.append( "\\f"); break;
105  case '\n': dest.append( "\\n"); break;
106  case '\r': dest.append( "\\r"); break;
107  case '\t': dest.append( "\\t"); break;
108  case '\v': dest.append( "\\v"); break;
109 
110  // single quote and misc. ASCII is OK
111  case '\'':
112  case ' ': case '!': case '#': case '$': case '%': case '&': case '(': case ')':
113  case '*': case '+': case ',': case '-': case '.': case '/':
114  case '0': case '1': case '2': case '3': case '4': case '5': case '6': case '7': case '8': case '9':
115  case ':': case ';': case '<': case '=': case '>': case '@':
116  case 'A': case 'B': case 'C': case 'D': case 'E': case 'F': case 'G': case 'H': case 'I': case 'J':
117  case 'K': case 'L': case 'M': case 'N': case 'O': case 'P': case 'Q': case 'R': case 'S': case 'T':
118  case 'U': case 'V': case 'W': case 'X': case 'Y': case 'Z':
119  case '[': case ']': case '^': case '_': case '`':
120  case 'a': case 'b': case 'c': case 'd': case 'e': case 'f': case 'g': case 'h': case 'i': case 'j':
121  case 'k': case 'l': case 'm': case 'n': case 'o': case 'p': case 'q': case 'r': case 's': case 't':
122  case 'u': case 'v': case 'w': case 'x': case 'y': case 'z':
123  case '{': case '|': case '}': case '~':
124  dest.append(&c, 1);
125  break;
126 
127  // use shortest octal escape for everything else
128  default:
129  dest.append("\\");
130  char dg[3];
131  dg[0] = '0' + ((c >> 6) & 3);
132  dg[1] = '0' + ((c >> 3) & 7);
133  dg[2] = '0' + ((c ) & 7);
134  int start = (dg[0] == '0' ? (dg[1] == '0' ? 2 : 1) : 0);
135  dest.append( dg+start, 3-start );
136  }
137  }
138  dest.append("\"");
139  needs_comma = true;
140  row++;
141  }
142  std::cerr << "\n";
143  return;
144 }

146 struct egenesis_info
147 {
148  fc::optional< genesis_state_type > genesis;
149  fc::optional< chain_id_type > chain_id;
150  fc::optional< std::string > genesis_json;
151  fc::optional< fc::sha256 > genesis_json_hash;
152  fc::optional< std::string > genesis_json_array;
153  int genesis_json_array_width,
154  genesis_json_array_height;
155 
156  void fillin()
157  {
158  // must specify either genesis_json or genesis
159  if( genesis.valid() )
160  {
161  if( !genesis_json.valid() )
162  // If genesis_json not exist, generate from genesis
163  genesis_json = fc::json::to_string( *genesis );
164  }
165  else if( genesis_json.valid() )
166  {
167  // If genesis not exist, generate from genesis_json
168  genesis = fc::json::from_string( *genesis_json ).as< genesis_state_type >( 20 );
169  }
170  else
171  {
172  // Neither genesis nor genesis_json exists, crippled
173  std::cerr << "embed_genesis: Need genesis or genesis_json\n";
174  exit(1);
175  }
176  // init genesis_json_hash from genesis_json
177  if( !genesis_json_hash.valid() )
178  genesis_json_hash = fc::sha256::hash( *genesis_json );
179  // init chain_id from genesis_json_hash
180  if( !chain_id.valid() )
181  chain_id = genesis_json_hash;
182  // init genesis_json_array from genesis_json
183  if( !genesis_json_array.valid() )
184  {
185  genesis_json_array = std::string();
186  // TODO: gzip
187  int width = 40;
188  convert_to_c_array( *genesis_json, *genesis_json_array, width );
189  int height = (genesis_json->length() + width-1) / width;
190  genesis_json_array_width = width;
191  genesis_json_array_height = height;
192  }
193  }
194 };

263  }
264
