252  // Note:
253  // Different type of object_id<T> objects could become identical after packed.
254  // For example, both `account_id_type a=1.2.0` and `asset_id_type b=1.3.0` will become `0` after packed.
255  // In order to avoid collision, we don't use a template function here, instead, we implicitly convert all
256  // object IDs to `object_id_type` when subscribing.
257  //
258  // If need to subscribe to other data types, override this function with the types as parameter.
259  // For example, we had a `get_subscription_key( const public_key_type& item )` function here, which was
260  // removed lately since we no longer subscribe to public keys.
261  vector<char> get_subscription_key( const object_id_type& item )const
262  {
263  return fc::raw::pack(item);
264  }
265 
266  template<typename T>
267  void subscribe_to_item( const T& item )const
268  {
269  if( !_subscribe_callback )
270  return;
271 
272  vector<char> key = get_subscription_key( item );
273  if( !_subscribe_filter.contains( key.data(), key.size() ) )
274  {
275  _subscribe_filter.insert( key.data(), key.size() );
276  }
277  }
278 
279  template<typename T>
280  bool is_subscribed_to_item( const T& item )const
281  {
282  if( !_subscribe_callback )
283  return false;
284 
285  vector<char> key = get_subscription_key( item );
286  return _subscribe_filter.contains( key.data(), key.size() );
287  }

309  template<typename T>
310  void enqueue_if_subscribed_to_market(const object* obj, market_queue_type& queue, bool full_object=true)
311  {
312  const T* order = dynamic_cast<const T*>(obj);
313  FC_ASSERT( order != nullptr);
314

316

320  }
