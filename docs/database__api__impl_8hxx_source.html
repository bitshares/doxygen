295  template<class LP>
296  extended_liquidity_pool_object extend_liquidity_pool( LP&& a, bool with_stats )const
297  {
298  liquidity_pool_id_type id = a.id;
299  extended_liquidity_pool_object result = extended_liquidity_pool_object( std::forward<LP>( a ) );
300  if( with_stats && _app_options && _app_options->has_market_history_plugin )
301  {
302  liquidity_pool_ticker_id_type ticker_id( id.instance );
303  const liquidity_pool_ticker_object* ticker = _db.find<liquidity_pool_ticker_object>( ticker_id );

306  }
307  return result;
308  }

310  // template function to reduce duplicate code
311  template <typename X>
312  vector<extended_liquidity_pool_object> get_liquidity_pools_by_asset_x(
313  std::string asset_symbol_or_id,
314  optional<uint32_t> olimit,
315  optional<liquidity_pool_id_type> ostart_id,
316  optional<bool> with_statistics )const
317  {
318  uint32_t limit = olimit.valid() ? *olimit : 101;
319 
320  FC_ASSERT( _app_options, "Internal error" );
321  const auto configured_limit = _app_options->api_limit_get_liquidity_pools;
322  FC_ASSERT( limit <= configured_limit,
323  "limit can not be greater than ${configured_limit}",
324  ("configured_limit", configured_limit) );
325 
326  bool with_stats = ( with_statistics.valid() && *with_statistics );
327 
328  vector<extended_liquidity_pool_object> results;
329 
330  const asset_id_type asset_id = get_asset_from_string(asset_symbol_or_id)->id;
331 
332  liquidity_pool_id_type start_id = ostart_id.valid() ? *ostart_id : liquidity_pool_id_type();
333 
334  const auto& idx = _db.get_index_type<liquidity_pool_index>().indices().get<X>();
335  auto lower_itr = idx.lower_bound( std::make_tuple( asset_id, start_id ) );
336  auto upper_itr = idx.upper_bound( asset_id );
337 
338  results.reserve( limit );
339  for ( ; lower_itr != upper_itr && results.size() < limit; ++lower_itr )
340  {
341  results.emplace_back( extend_liquidity_pool( *lower_itr, with_stats ) );
342  }
343 
344  return results;
345  }

365  template <typename OBJ_TYPE, typename OBJ_ID_TYPE, typename INDEX_TYPE, typename T, typename... X >
366  vector<OBJ_TYPE> get_objects_by_x(
367  T application_options::* app_opt_member_ptr,
368  const INDEX_TYPE& idx,
369  const optional<uint32_t>& olimit,
370  const optional<OBJ_ID_TYPE>& ostart_id,
371  X... x ) const
372  {
373  uint64_t limit = olimit.valid() ? *olimit : ( application_options::get_default().*app_opt_member_ptr );
374 
375  FC_ASSERT( _app_options, "Internal error" );
376  const auto configured_limit = _app_options->*app_opt_member_ptr;
377  FC_ASSERT( limit <= configured_limit,
378  "limit can not be greater than ${configured_limit}",
379  ("configured_limit", configured_limit) );
380 
381  vector<OBJ_TYPE> results;
382 
383  OBJ_ID_TYPE start_id = ostart_id.valid() ? *ostart_id : OBJ_ID_TYPE();
384 
385  auto lower_itr = idx.lower_bound( make_tuple_if_multiple( x..., start_id ) );
386  auto upper_itr = call_end_or_upper_bound( idx, x... );
387 
388  results.reserve( limit );
389  for ( ; lower_itr != upper_itr && results.size() < limit; ++lower_itr )
390  {
391  results.emplace_back( *lower_itr );
392  }
393 
394  return results;
395  }

412  // Note:
413  // Different type of object_id<T> objects could become identical after packed.
414  // For example, both `account_id_type a=1.2.0` and `asset_id_type b=1.3.0` will become `0` after packed.
415  // In order to avoid collision, we don't use a template function here, instead, we implicitly convert all
416  // object IDs to `object_id_type` when subscribing.
417  //
418  // If need to subscribe to other data types, override this function with the types as parameter.
419  // For example, we had a `get_subscription_key( const public_key_type& item )` function here, which was
420  // removed lately since we no longer subscribe to public keys.
421  vector<char> get_subscription_key( const object_id_type& item )const
422  {
423  return fc::raw::pack(item);
424  }
425 
426  template<typename T>
427  void subscribe_to_item( const T& item )const
428  {
429  if( !_subscribe_callback )
430  return;
431 
432  vector<char> key = get_subscription_key( item );
433  if( !_subscribe_filter.contains( key.data(), key.size() ) )
434  {
435  _subscribe_filter.insert( key.data(), key.size() );
436  }
437  }
438 
439  template<typename T>
440  bool is_subscribed_to_item( const T& item )const
441  {
442  if( !_subscribe_callback )
443  return false;
444 
445  vector<char> key = get_subscription_key( item );
446  return _subscribe_filter.contains( key.data(), key.size() );
447  }
