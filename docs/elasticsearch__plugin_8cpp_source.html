107 bool elasticsearch_plugin_impl::update_account_histories( const signed_block& b )
108 {
109  checkState(b.timestamp);
110  index_name = graphene::utilities::generateIndexName(b.timestamp, _elasticsearch_index_prefix);
111 
112  graphene::chain::database& db = database();
113  const vector<optional< operation_history_object > >& hist = db.get_applied_operations();
114  bool is_first = true;
115  auto skip_oho_id = [&is_first,&db,this]() {
116  if( is_first && db._undo_db.enabled() ) // this ensures that the current id is rolled back on undo
117  {
118  db.remove( db.create<operation_history_object>( []( operation_history_object& obj) {} ) );
119  is_first = false;
120  }
121  else
122  _oho_index->use_next_id();
123  };
124  for( const optional< operation_history_object >& o_op : hist ) {
125  optional <operation_history_object> oho;
126 
127  auto create_oho = [&]() {
128  is_first = false;
129  return optional<operation_history_object>(
130  db.create<operation_history_object>([&](operation_history_object &h) {
131  if (o_op.valid())
132  {
133  h.op = o_op->op;
134  h.result = o_op->result;
135  h.block_num = o_op->block_num;
136  h.trx_in_block = o_op->trx_in_block;
137  h.op_in_trx = o_op->op_in_trx;
138  h.virtual_op = o_op->virtual_op;
139  }
140  }));
141  };
142 
143  if( !o_op.valid() ) {
144  skip_oho_id();
145  continue;
146  }
147  oho = create_oho();
148 
149  // populate what we can before impacted loop
150  getOperationType(oho);
151  doOperationHistory(oho);
152  doBlock(oho->trx_in_block, b);
153  if(_elasticsearch_visitor)
154  doVisitor(oho);
155 
156  const operation_history_object& op = *o_op;
157 
158  // get the set of accounts this operation applies to
159  flat_set<account_id_type> impacted;
160  vector<authority> other;
161  // fee_payer is added here
162  operation_get_required_authorities( op.op, impacted, impacted, other,
163  MUST_IGNORE_CUSTOM_OP_REQD_AUTHS( db.head_block_time() ) );
164 
165  if( op.op.is_type< account_create_operation >() )
166  impacted.insert( op.result.get<object_id_type>() );
167  else
168  operation_get_impacted_accounts( op.op, impacted,
169  MUST_IGNORE_CUSTOM_OP_REQD_AUTHS( db.head_block_time() ) );
170 
171  for( auto& a : other )
172  for( auto& item : a.account_auths )
173  impacted.insert( item.first );
174 
175  for( auto& account_id : impacted )
176  {
177  if(!add_elasticsearch( account_id, oho, b.block_num() ))
178  {
179  elog( "Error adding data to Elastic Search: block num ${b}, account ${a}, data ${d}",
180  ("b",b.block_num()) ("a",account_id) ("d", oho) );
181  return false;
182  }
183  }
184  }
185  // we send bulk at end of block when we are in sync for better real time client experience
186  if(is_sync)
187  {
188  populateESstruct();
189  if(es.bulk_lines.size() > 0)
190  {
191  prepare.clear();
192  if(!graphene::utilities::SendBulk(std::move(es)))
193  {
194  // Note: although called with `std::move()`, `es` is not updated in `SendBulk()`
195  elog( "Error sending ${n} lines of bulk data to Elastic Search, the first lines are:",
196  ("n",es.bulk_lines.size()) );
197  for( size_t i = 0; i < es.bulk_lines.size() && i < 10; ++i )
198  {
199  edump( (es.bulk_lines[i]) );
200  }
201  return false;
202  }
203  else
204  bulk_lines.clear();
205  }
206  }
207 
208  if(bulk_lines.size() != limit_documents)
209  bulk_lines.reserve(limit_documents);
210 
211  return true;
212 }

301 bool elasticsearch_plugin_impl::add_elasticsearch( const account_id_type account_id,
302  const optional <operation_history_object>& oho,
303  const uint32_t block_number)
304 {
305  const auto &stats_obj = getStatsObject(account_id);
306  const auto &ath = addNewEntry(stats_obj, account_id, oho);
307  growStats(stats_obj, ath);
308  if(block_number > _elasticsearch_start_es_after_block) {
309  createBulkLine(ath);
310  prepareBulk(ath.id);
311  }
312  cleanObjects(ath.id, account_id);
313 
314  if (curl && bulk_lines.size() >= limit_documents) { // we are in bulk time, ready to add data to elasticsearech
315  prepare.clear();
316  populateESstruct();
317  if(!graphene::utilities::SendBulk(std::move(es)))
318  {
319  // Note: although called with `std::move()`, `es` is not updated in `SendBulk()`
320  elog( "Error sending ${n} lines of bulk data to Elastic Search, the first lines are:",
321  ("n",es.bulk_lines.size()) );
322  for( size_t i = 0; i < es.bulk_lines.size() && i < 10; ++i )
323  {
324  edump( (es.bulk_lines[i]) );
325  }
326  return false;
327  }
328  else
329  bulk_lines.clear();
330  }
331 
332  return true;
333 }

393 void elasticsearch_plugin_impl::cleanObjects(const account_transaction_history_id_type& ath_id, const account_id_type& account_id)
394 {
395  graphene::chain::database& db = database();
396  // remove everything except current object from ath
397  const auto &his_idx = db.get_index_type<account_transaction_history_index>();
398  const auto &by_seq_idx = his_idx.indices().get<by_seq>();
399  auto itr = by_seq_idx.lower_bound(boost::make_tuple(account_id, 0));
400  if (itr != by_seq_idx.end() && itr->account == account_id && itr->id != ath_id) {
401  // if found, remove the entry
402  const auto remove_op_id = itr->operation_id;
403  const auto itr_remove = itr;
404  ++itr;
405  db.remove( *itr_remove );
406  // modify previous node's next pointer
407  // this should be always true, but just have a check here
408  if( itr != by_seq_idx.end() && itr->account == account_id )
409  {
410  db.modify( *itr, [&]( account_transaction_history_object& obj ){
411  obj.next = account_transaction_history_id_type();
412  });
413  }
414  // do the same on oho
415  const auto &by_opid_idx = his_idx.indices().get<by_opid>();
416  if (by_opid_idx.find(remove_op_id) == by_opid_idx.end()) {
417  db.remove(remove_op_id(db));
418  }
419  }

483 void elasticsearch_plugin::plugin_initialize(const boost::program_options::variables_map& options)
484 {

486  database().add_index< primary_index< account_transaction_history_index > >();

498  my->_elasticsearch_visitor = options["elasticsearch-visitor"].as<bool>();
499  }
500  if (options.count("elasticsearch-basic-auth") > 0) {
501  my->_elasticsearch_basic_auth = options["elasticsearch-basic-auth"].as<std::string>();
502  }
503  if (options.count("elasticsearch-index-prefix") > 0) {
504  my->_elasticsearch_index_prefix = options["elasticsearch-index-prefix"].as<std::string>();
505  }
506  if (options.count("elasticsearch-operation-object") > 0) {
507  my->_elasticsearch_operation_object = options["elasticsearch-operation-object"].as<bool>();
508  }
509  if (options.count("elasticsearch-start-es-after-block") > 0) {
510  my->_elasticsearch_start_es_after_block = options["elasticsearch-start-es-after-block"].as<uint32_t>();
511  }
512  if (options.count("elasticsearch-operation-string") > 0) {
513  my->_elasticsearch_operation_string = options["elasticsearch-operation-string"].as<bool>();
514  }
515  if (options.count("elasticsearch-mode") > 0) {
516  const auto option_number = options["elasticsearch-mode"].as<uint16_t>();
517  if(option_number > mode::all)
518  FC_THROW_EXCEPTION(graphene::chain::plugin_exception, "Elasticsearch mode not valid");
519  my->_elasticsearch_mode = static_cast<mode>(options["elasticsearch-mode"].as<uint16_t>());
520  }
521 
522  if(my->_elasticsearch_mode != mode::only_query) {
523  if (my->_elasticsearch_mode == mode::all && !my->_elasticsearch_operation_string)
524  FC_THROW_EXCEPTION(graphene::chain::plugin_exception,
525  "If elasticsearch-mode is set to all then elasticsearch-operation-string need to be true");
526 
527  database().applied_block.connect([this](const signed_block &b) {
528  if (!my->update_account_histories(b))
529  FC_THROW_EXCEPTION(graphene::chain::plugin_exception,
530  "Error populating ES database, we are going to keep trying.");
531  });
532  }

536 {

538  es.curl = my->curl;
539  es.elasticsearch_url = my->_elasticsearch_node_url;
