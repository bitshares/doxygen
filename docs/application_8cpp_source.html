166  c->set_session_data( wsc );
167 
168  std::string username = "*";
169  std::string password = "*";
170 
171  // Try to extract login information from "Authorization" header if present
172  std::string auth = c->get_request_header("Authorization");
173  if( boost::starts_with(auth, "Basic ") ) {
174 
175  FC_ASSERT( auth.size() > 6 );
176  auto user_pass = fc::base64_decode(auth.substr(6));
177 
178  std::vector<std::string> parts;
179  boost::split( parts, user_pass, boost::is_any_of(":") );
180 
181  FC_ASSERT(parts.size() == 2);
182 
183  username = parts[0];
184  password = parts[1];
185  }
186 
187  login->login(username, password);

442  auto genesis = fc::json::from_string( egenesis_json ).as<graphene::chain::genesis_state_type>( 20 );
443  genesis.initial_chain_id = fc::sha256::hash( egenesis_json );
444  return genesis;

449  _chain_db->wipe(_data_dir / "blockchain", true);
450 
451  flat_map<uint32_t,block_id_type> loaded_checkpoints;

472  try
473  {
474  // these flags are used in open() only, i. e. during replay
475  uint32_t skip;
476  if( _options->count("revalidate-blockchain") ) // see also handle_block()
477  {
478  if( !loaded_checkpoints.empty() )
479  wlog( "Warning - revalidate will not validate before last checkpoint" );
480  if( _options->count("force-validate") )
481  skip = graphene::chain::database::skip_nothing;
482  else
483  skip = graphene::chain::database::skip_transaction_signatures;
484  }
485  else // no revalidate, skip most checks
486  skip = graphene::chain::database::skip_witness_signature |
487  graphene::chain::database::skip_block_size_check |
488  graphene::chain::database::skip_merkle_check |
489  graphene::chain::database::skip_transaction_signatures |

555 bool application_impl::handle_block(const graphene::net::block_message& blk_msg, bool sync_mode,
556  std::vector<fc::uint160_t>& contained_transaction_message_ids)
557 { try {
558 
559  auto latency = fc::time_point::now() - blk_msg.block.timestamp;
560  if (!sync_mode || blk_msg.block.block_num() % 10000 == 0)
561  {
562  const auto& witness = blk_msg.block.witness(*_chain_db);
563  const auto& witness_account = witness.witness_account(*_chain_db);
564  auto last_irr = _chain_db->get_dynamic_global_properties().last_irreversible_block_num;
565  ilog("Got block: #${n} ${bid} time: ${t} transaction(s): ${x} latency: ${l} ms from: ${w} irreversible: ${i} (-${d})",
566  ("t",blk_msg.block.timestamp)
567  ("n", blk_msg.block.block_num())
568  ("bid", blk_msg.block.id())
569  ("x", blk_msg.block.transactions.size())
570  ("l", (latency.count()/1000))
571  ("w",witness_account.name)
572  ("i",last_irr)("d",blk_msg.block.block_num()-last_irr) );
573  }
574  GRAPHENE_ASSERT( latency.count()/1000 > -5000,
575  graphene::net::block_timestamp_in_future_exception,
576  "Rejecting block with timestamp in the future", );
577 
578  try {
579  const uint32_t skip = (_is_block_producer | _force_validate) ?
580  database::skip_nothing : database::skip_transaction_signatures;
581  bool result = valve.do_serial( [this,&blk_msg,skip] () {
582  _chain_db->precompute_parallel( blk_msg.block, skip ).wait();
583  }, [this,&blk_msg,skip] () {
584  // TODO: in the case where this block is valid but on a fork that's too old for us to switch to,
585  // you can help the network code out by throwing a block_older_than_undo_history exception.
586  // when the net code sees that, it will stop trying to push blocks from that chain, but
587  // leave that peer connected so that they can get sync blocks from us
588  return _chain_db->push_block( blk_msg.block, skip );
589  });
590 
591  // the block was accepted, so we now know all of the transactions contained in the block
592  if (!sync_mode)
593  {
594  // if we're not in sync mode, there's a chance we will be seeing some transactions
595  // included in blocks before we see the free-floating transaction itself. If that
596  // happens, there's no reason to fetch the transactions, so construct a list of the
597  // transaction message ids we no longer need.
598  // during sync, it is unlikely that we'll see any old
599  contained_transaction_message_ids.reserve( contained_transaction_message_ids.size()
600  + blk_msg.block.transactions.size() );
601  for (const processed_transaction& transaction : blk_msg.block.transactions)
602  {
603  graphene::net::trx_message transaction_message(transaction);
604  contained_transaction_message_ids.emplace_back(graphene::net::message(transaction_message).id());
605  }
606  }
607 
608  return result;
609  } catch ( const graphene::chain::unlinkable_block_exception& e ) {
610  // translate to a graphene::net exception
611  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
612  FC_THROW_EXCEPTION( graphene::net::unlinkable_block_exception,
613  "Error when pushing block:\n${e}",
614  ("e", e.to_detail_string()) );
615  } catch( const fc::exception& e ) {
616  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
617  throw;
618  }
619 
620  if( !_is_finished_syncing && !sync_mode )
621  {
622  _is_finished_syncing = true;
623  _self->syncing_finished();
624  }
625 } FC_CAPTURE_AND_RETHROW( (blk_msg)(sync_mode) ) return false; }

665 std::vector<item_hash_t> application_impl::get_block_ids(const std::vector<item_hash_t>& blockchain_synopsis,
666  uint32_t& remaining_item_count,
667  uint32_t limit)
668 { try {
669  vector<block_id_type> result;
670  remaining_item_count = 0;
671  if( _chain_db->head_block_num() == 0 )
672  return result;
673 
674  result.reserve(limit);
675  block_id_type last_known_block_id;
676 
677  if (blockchain_synopsis.empty() ||
678  (blockchain_synopsis.size() == 1 && blockchain_synopsis[0] == block_id_type()))
679  {
680  // peer has sent us an empty synopsis meaning they have no blocks.
681  // A bug in old versions would cause them to send a synopsis containing block 000000000
682  // when they had an empty blockchain, so pretend they sent the right thing here.
683 
684  // do nothing, leave last_known_block_id set to zero
685  }
686  else
687  {
688  bool found_a_block_in_synopsis = false;
689  for (const item_hash_t& block_id_in_synopsis : boost::adaptors::reverse(blockchain_synopsis))
690  if (block_id_in_synopsis == block_id_type() ||
691  (_chain_db->is_known_block(block_id_in_synopsis) && is_included_block(block_id_in_synopsis)))
692  {
693  last_known_block_id = block_id_in_synopsis;
694  found_a_block_in_synopsis = true;
695  break;
696  }
697  if (!found_a_block_in_synopsis)
698  FC_THROW_EXCEPTION( graphene::net::peer_is_on_an_unreachable_fork,
699  "Unable to provide a list of blocks starting at any of the blocks in peer's synopsis" );
700  }
701  for( uint32_t num = block_header::num_from_id(last_known_block_id);
702  num <= _chain_db->head_block_num() && result.size() < limit;
703  ++num )
704  if( num > 0 )
705  result.push_back(_chain_db->get_block_id_for_num(num));
706 
707  if( !result.empty() && block_header::num_from_id(result.back()) < _chain_db->head_block_num() )
708  remaining_item_count = _chain_db->head_block_num() - block_header::num_from_id(result.back());
709 
710  return result;
711 } FC_CAPTURE_AND_RETHROW( (blockchain_synopsis)(remaining_item_count)(limit) ) }

795 std::vector<item_hash_t> application_impl::get_blockchain_synopsis(const item_hash_t& reference_point,
796  uint32_t number_of_blocks_after_reference_point)
797 { try {
798  std::vector<item_hash_t> synopsis;
799  synopsis.reserve(30);
800  uint32_t high_block_num;
801  uint32_t non_fork_high_block_num;
802  uint32_t low_block_num = _chain_db->last_non_undoable_block_num();
803  std::vector<block_id_type> fork_history;
804 
805  if (reference_point != item_hash_t())
806  {
807  // the node is asking for a summary of the block chain up to a specified
808  // block, which may or may not be on a fork
809  // for now, assume it's not on a fork
810  if (is_included_block(reference_point))
811  {
812  // reference_point is a block we know about and is on the main chain
813  uint32_t reference_point_block_num = block_header::num_from_id(reference_point);
814  assert(reference_point_block_num > 0);
815  high_block_num = reference_point_block_num;
816  non_fork_high_block_num = high_block_num;
817 
818  if (reference_point_block_num < low_block_num)
819  {
820  // we're on the same fork (at least as far as reference_point) but we've passed
821  // reference point and could no longer undo that far if we diverged after that
822  // block. This should probably only happen due to a race condition where
823  // the network thread calls this function, and then immediately pushes a bunch of blocks,
824  // then the main thread finally processes this function.
825  // with the current framework, there's not much we can do to tell the network
826  // thread what our current head block is, so we'll just pretend that
827  // our head is actually the reference point.
828  // this *may* enable us to fetch blocks that we're unable to push, but that should
829  // be a rare case (and correctly handled)
830  low_block_num = reference_point_block_num;
831  }
832  }
833  else
834  {
835  // block is a block we know about, but it is on a fork
836  try
837  {
838  fork_history = _chain_db->get_block_ids_on_fork(reference_point);
839  // returns a vector where the last element is the common ancestor with the preferred chain,
840  // and the first element is the reference point you passed in
841  assert(fork_history.size() >= 2);
842 
843  if( fork_history.front() != reference_point )
844  {
845  edump( (fork_history)(reference_point) );
846  assert(fork_history.front() == reference_point);
847  }
848  block_id_type last_non_fork_block = fork_history.back();
849  fork_history.pop_back(); // remove the common ancestor
850  boost::reverse(fork_history);
851 
852  if (last_non_fork_block == block_id_type()) // if the fork goes all the way back to genesis (does graphene's fork db allow this?)
853  non_fork_high_block_num = 0;
854  else
855  non_fork_high_block_num = block_header::num_from_id(last_non_fork_block);
856 
857  high_block_num = non_fork_high_block_num + fork_history.size();
858  assert(high_block_num == block_header::num_from_id(fork_history.back()));
859  }
860  catch (const fc::exception& e)
861  {
862  // unable to get fork history for some reason. maybe not linked?
863  // we can't return a synopsis of its chain
864  elog( "Unable to construct a blockchain synopsis for reference hash ${hash}: ${exception}",
865  ("hash", reference_point)("exception", e) );
866  throw;
867  }
868  if (non_fork_high_block_num < low_block_num)
869  {
870  wlog("Unable to generate a usable synopsis because the peer we're generating it for forked too long ago "
871  "(our chains diverge after block #${non_fork_high_block_num} but only undoable to block #${low_block_num})",
872  ("low_block_num", low_block_num)
873  ("non_fork_high_block_num", non_fork_high_block_num));
874  FC_THROW_EXCEPTION(graphene::net::block_older_than_undo_history, "Peer is are on a fork I'm unable to switch to");
875  }
876  }
877  }
878  else
879  {
880  // no reference point specified, summarize the whole block chain
881  high_block_num = _chain_db->head_block_num();
882  non_fork_high_block_num = high_block_num;
883  if (high_block_num == 0)
884  return synopsis; // we have no blocks
885  }
886 
887  if( low_block_num == 0)
888  low_block_num = 1;
889 
890  // at this point:
891  // low_block_num is the block before the first block we can undo,
892  // non_fork_high_block_num is the block before the fork (if the peer is on a fork, or otherwise it is the same as high_block_num)
893  // high_block_num is the block number of the reference block, or the end of the chain if no reference provided
894 
895  // true_high_block_num is the ending block number after the network code appends any item ids it
896  // knows about that we don't
897  uint32_t true_high_block_num = high_block_num + number_of_blocks_after_reference_point;
898  do
899  {
900  // for each block in the synopsis, figure out where to pull the block id from.
901  // if it's <= non_fork_high_block_num, we grab it from the main blockchain;
902  // if it's not, we pull it from the fork history
903  if (low_block_num <= non_fork_high_block_num)
904  synopsis.push_back(_chain_db->get_block_id_for_num(low_block_num));
905  else
906  synopsis.push_back(fork_history[low_block_num - non_fork_high_block_num - 1]);
907  low_block_num += (true_high_block_num - low_block_num + 2) / 2;
908  }
909  while (low_block_num <= high_block_num);
910 
911  //idump((synopsis));
912  return synopsis;
913 } FC_CAPTURE_AND_RETHROW() }
