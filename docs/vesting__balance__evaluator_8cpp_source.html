96  const vesting_balance_object& vbo = d.create< vesting_balance_object >( [&]( vesting_balance_object& obj )
97  {
98  //WARNING: The logic to create a vesting balance object is replicated in vesting_balance_worker_type::initializer::init.

100  obj.owner = op.owner;
101  obj.balance = op.amount;
102  op.policy.visit( init_policy_visitor( obj.policy, op.amount.amount, now ) );
