83 bool es_objects_plugin_impl::genesis()
84 {
85  ilog("elasticsearch OBJECTS: inserting data from genesis");
86 
87  graphene::chain::database &db = _self.database();
88 
89  block_number = db.head_block_num();
90  block_time = db.head_block_time();
91 
92  if (_es_objects_accounts) {
93  auto &index_accounts = db.get_index(1, 2);
94  index_accounts.inspect_all_objects([this, &db](const graphene::db::object &o) {
95  auto obj = db.find_object(o.id);
96  auto a = static_cast<const account_object *>(obj);
97  prepareTemplate<account_object>(*a, "account");
98  });
99  }
100  if (_es_objects_assets) {
101  auto &index_assets = db.get_index(1, 3);
102  index_assets.inspect_all_objects([this, &db](const graphene::db::object &o) {
103  auto obj = db.find_object(o.id);
104  auto a = static_cast<const asset_object *>(obj);
105  prepareTemplate<asset_object>(*a, "asset");
106  });
107  }
108  if (_es_objects_balances) {
109  auto &index_balances = db.get_index(2, 5);
110  index_balances.inspect_all_objects([this, &db](const graphene::db::object &o) {
111  auto obj = db.find_object(o.id);
112  auto b = static_cast<const account_balance_object *>(obj);
113  prepareTemplate<account_balance_object>(*b, "balance");
114  });
115  }
116 
117  graphene::utilities::ES es;
118  es.curl = curl;
119  es.bulk_lines = bulk;
120  es.elasticsearch_url = _es_objects_elasticsearch_url;
121  es.auth = _es_objects_auth;
122  if (!graphene::utilities::SendBulk(std::move(es)))
123  FC_THROW_EXCEPTION(graphene::chain::plugin_exception, "Error inserting genesis data.");
124  else
125  bulk.clear();
126

130 bool es_objects_plugin_impl::index_database(const vector<object_id_type>& ids, std::string action)
131 {
132  graphene::chain::database &db = _self.database();
133 
134  block_time = db.head_block_time();
135  block_number = db.head_block_num();
136 
137  if(block_number > _es_objects_start_es_after_block) {
138 
139  // check if we are in replay or in sync and change number of bulk documents accordingly
140  uint32_t limit_documents = 0;
141  if ((fc::time_point::now() - block_time) < fc::seconds(30))
142  limit_documents = _es_objects_bulk_sync;
143  else
144  limit_documents = _es_objects_bulk_replay;
145 
146 
147  for (auto const &value: ids) {
148  if (value.is<proposal_object>() && _es_objects_proposals) {
149  auto obj = db.find_object(value);
150  auto p = static_cast<const proposal_object *>(obj);
151  if (p != nullptr) {
152  if (action == "delete")
153  remove_from_database(p->id, "proposal");
154  else
155  prepareTemplate<proposal_object>(*p, "proposal");
156  }
157  } else if (value.is<account_object>() && _es_objects_accounts) {
158  auto obj = db.find_object(value);
159  auto a = static_cast<const account_object *>(obj);
160  if (a != nullptr) {
161  if (action == "delete")
162  remove_from_database(a->id, "account");
163  else
164  prepareTemplate<account_object>(*a, "account");
165  }
166  } else if (value.is<asset_object>() && _es_objects_assets) {
167  auto obj = db.find_object(value);
168  auto a = static_cast<const asset_object *>(obj);
169  if (a != nullptr) {
170  if (action == "delete")
171  remove_from_database(a->id, "asset");
172  else
173  prepareTemplate<asset_object>(*a, "asset");
174  }
175  } else if (value.is<account_balance_object>() && _es_objects_balances) {
176  auto obj = db.find_object(value);
177  auto b = static_cast<const account_balance_object *>(obj);
178  if (b != nullptr) {
179  if (action == "delete")
180  remove_from_database(b->id, "balance");
181  else
182  prepareTemplate<account_balance_object>(*b, "balance");
183  }
184  } else if (value.is<limit_order_object>() && _es_objects_limit_orders) {
185  auto obj = db.find_object(value);
186  auto l = static_cast<const limit_order_object *>(obj);
187  if (l != nullptr) {
188  if (action == "delete")
189  remove_from_database(l->id, "limitorder");
190  else
191  prepareTemplate<limit_order_object>(*l, "limitorder");
192  }
193  } else if (value.is<asset_bitasset_data_object>() && _es_objects_asset_bitasset) {
194  auto obj = db.find_object(value);
195  auto ba = static_cast<const asset_bitasset_data_object *>(obj);
196  if (ba != nullptr) {
197  if (action == "delete")
198  remove_from_database(ba->id, "bitasset");
199  else
200  prepareTemplate<asset_bitasset_data_object>(*ba, "bitasset");
201  }
202  }
203  }
204 
205  if (curl && bulk.size() >= limit_documents) { // we are in bulk time, ready to add data to elasticsearech
206 
207  graphene::utilities::ES es;
208  es.curl = curl;
209  es.bulk_lines = bulk;
210  es.elasticsearch_url = _es_objects_elasticsearch_url;
211  es.auth = _es_objects_auth;
212 
213  if (!graphene::utilities::SendBulk(std::move(es)))
214  return false;
215  else
216  bulk.clear();
217  }
218  }
219 
220  return true;
221 }

223 void es_objects_plugin_impl::remove_from_database( object_id_type id, std::string index)
224 {
225  if(_es_objects_keep_only_current)
226  {
227  fc::mutable_variant_object delete_line;
228  delete_line["_id"] = string(id);
229  delete_line["_index"] = _es_objects_index_prefix + index;
230  delete_line["_type"] = "data";
231  fc::mutable_variant_object final_delete_line;
232  final_delete_line["delete"] = delete_line;
233  prepare.push_back(fc::json::to_string(final_delete_line));
234  std::move(prepare.begin(), prepare.end(), std::back_inserter(bulk));
235  prepare.clear();
236  }

239 template<typename T>
240 void es_objects_plugin_impl::prepareTemplate(T blockchain_object, string index_name)
241 {
242  fc::mutable_variant_object bulk_header;
243  bulk_header["_index"] = _es_objects_index_prefix + index_name;
244  bulk_header["_type"] = "data";
245  if(_es_objects_keep_only_current)
246  {
247  bulk_header["_id"] = string(blockchain_object.id);
248  }
249 
250  adaptor_struct adaptor;
251  fc::variant blockchain_object_variant;
252  fc::to_variant( blockchain_object, blockchain_object_variant, GRAPHENE_NET_MAX_NESTED_OBJECTS );
253  fc::mutable_variant_object o = adaptor.adapt(blockchain_object_variant.get_object());
254 
255  o["object_id"] = string(blockchain_object.id);
256  o["block_time"] = block_time;
257  o["block_number"] = block_number;
258 
259  string data = fc::json::to_string(o, fc::json::legacy_generator);
260 
261  prepare = graphene::utilities::createBulk(bulk_header, std::move(data));
262  std::move(prepare.begin(), prepare.end(), std::back_inserter(bulk));
263  prepare.clear();
264 }
