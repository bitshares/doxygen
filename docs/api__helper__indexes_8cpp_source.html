180 void api_helper_indexes::plugin_startup()
181 {
182  ilog("api_helper_indexes: plugin_startup() begin");
183  amount_in_collateral_idx = database().add_secondary_index< primary_index<call_order_index>,
184  amount_in_collateral_index >();
185  for( const auto& call : database().get_index_type<call_order_index>().indices() )
186  amount_in_collateral_idx->object_inserted( call );
187 
188  auto& account_members = *database().add_secondary_index< primary_index<account_index>, account_member_index >();
189  for( const auto& account : database().get_index_type< account_index >().indices() )
190  account_members.object_inserted( account );
191 
192  auto& approvals = *database().add_secondary_index< primary_index<proposal_index>, required_approval_index >();
193  for( const auto& proposal : database().get_index_type< proposal_index >().indices() )
194  approvals.object_inserted( proposal );
195 
196  asset_in_liquidity_pools_idx = database().add_secondary_index< primary_index<liquidity_pool_index>,
197  asset_in_liquidity_pools_index >();
198  for( const auto& pool : database().get_index_type<liquidity_pool_index>().indices() )
199  asset_in_liquidity_pools_idx->object_inserted( pool );

201 }
